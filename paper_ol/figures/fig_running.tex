% Figure 1 — running example: 1-hop CN fails, 2-hop cohesive bridge works.
% \input{figures/fig_running} inside a figure environment.
\begin{tikzpicture}[
  every node/.style={font=\footnotesize},
  nd/.style={circle,draw,minimum size=13pt,inner sep=0pt},
  hi/.style={circle,draw,minimum size=14pt,inner sep=0pt,fill=gray!50},
  qry/.style={circle,draw,minimum size=16pt,inner sep=0pt,line width=0.9pt},
  brg/.style={orange!85!black,line width=1.7pt},
  x=1cm,y=1cm]

  % query nodes
  \node[qry,fill=blue!60,text=white] (u) at (0,1.9) {$u$};
  \node[qry,fill=red!70,text=white]  (v) at (5.0,0.0) {$v$};

  % u's direct neighbours
  \node[hi] (a) at (1.7,2.6) {$a$};
  \node[nd] (b) at (1.7,1.2) {$b$};
  \node[nd] (c) at (0.0,3.0) {};
  % v's direct neighbours
  \node[hi] (d) at (3.3,0.7) {$d$};
  \node[nd] (e) at (3.3,-0.8) {$e$};
  \node[nd] (f) at (5.0,1.3) {};

  % ordinary edges
  \draw (u)--(a); \draw (u)--(b); \draw (u)--(c);
  \draw (v)--(d); \draw (v)--(e); \draw (v)--(f);

  % the two 2-hop cohesive bridges (highlighted)
  \draw[brg] (a)--(d);
  \draw[brg] (b)--(e);

  % candidate (future) edge
  \draw[dashed,gray!70] (u)--(v);
  \node[gray!70,font=\small] at (2.5,1.15) {?};

  % core labels on the high-cohesion bridge nodes
  \node[font=\scriptsize,gray!55!black] at (1.7,3.15) {core\,$=5$};
  \node[font=\scriptsize,gray!55!black] at (3.95,0.7) {core\,$=4$};

  % annotation
  \node[font=\scriptsize,anchor=north] at (2.5,-1.25)
    {$\mathrm{CN}(u,v)=0$\quad\textcolor{orange!85!black}{$\mathrm{bridge}(u,v)=2$}};
\end{tikzpicture}
